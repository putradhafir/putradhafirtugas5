1. MULTIPLE INHERITANCE dengan MIXIN

Dart menggunakan Mixin untuk mencapai pewarisan ganda (multiple inheritance). Mixin memungkinkan sebuah class menggunakan fungsionalitas dari beberapa class lain tanpa hierarki inheritance yang rumit. Berbeda dengan Java atau C# yang hanya mendukung single inheritance, Dart memberikan fleksibilitas ini melalui keyword with. 


// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

mixin CanFly {
  void fly() {
    print('Can fly!');
  }
}

mixin CanSwim {
  void swim() {
    print('Can swim!');
  }
}

class Bird with CanFly {}

class Duck extends Bird with CanSwim {}

void main() {
  Bird bird = Bird();
  bird.fly();  // Output: Can fly!

  Duck duck = Duck();
  duck.fly();  // Output: Can fly!
  duck.swim(); // Output: Can swim!
}

2. MIXIN dengan STATE (Properti)

Mixin tidak hanya bisa berisi method, tetapi juga properti (state). Dengan menggunakan properti private dan getter, kita dapat menjaga state internal mixin agar tetap terkontrol. 


// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

mixin CanFly {
  bool _isFlying = false;
  bool get isFlying => _isFlying;
  
  void fly() {
    if (!_isFlying) {
      _isFlying = true;
      print('Started flying!');
    } else {
      print('Already flying!');
    }
  }
  
  void land() {
    if (_isFlying) {
      _isFlying = false;
      print('Landed!');
    } else {
      print('Already on the ground!');
    }
  }
}

mixin CanSwim {
  bool _isSwimming = false;
  bool get isSwimming => _isSwimming;
  
  void swim() {
    if (!_isSwimming) {
      _isSwimming = true;
      print('Started swimming!');
    } else {
      print('Already swimming!');
    }
  }
  
  void stopSwim() {
    if (_isSwimming) {
      _isSwimming = false;
      print('Stop swimming!');
    } else {
      print('Already on stop swimming!');
    }
  }
}

class Bird with CanFly {}
class Duck extends Bird with CanSwim {}
class Fish with CanSwim {}
class Airplane with CanFly {}

void main() {
  // Bird
  Bird bird = Bird();
  bird.fly();   // Started flying!
  bird.fly();   // Already flying!
  bird.land();  // Landed!
  print('Is Bird flying? : ${bird.isFlying}'); // false
  
  // Duck
  Duck duck = Duck();
  duck.fly();   // Started flying!
  duck.swim();  // Started swimming!
  duck.stopSwim(); // Stop swimming!
  
  // Fish
  Fish fish = Fish();
  fish.swim();  // Started swimming!
  fish.stopSwim(); // Stop swimming!
  
  // Airplane
  Airplane airplane = Airplane();
  airplane.fly();  // Started flying!
  airplane.land(); // Landed!
}

3. SEALED CLASS - Hierarki Tertutup

Sealed class adalah class yang tidak bisa di-extend atau di-implement dari luar library tempatnya didefinisikan. Cocok untuk pattern matching yang exhaustive dan memastikan semua kemungkinan subclass diketahui saat compile-time.

📌 Fitur Sealed Class:
✅ Tidak bisa di-extend/implement dari luar library
✅ Compiler tahu semua kemungkinan subclass (exhaustive checking)
✅ Perfect untuk state management dengan switch expression 

// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

// libsealed.dart
sealed class ApiResponse {}

class Success extends ApiResponse {
  final String message;
  Success(this.message);
}

class Error extends ApiResponse {
  final String errorMessage;
  Error(this.errorMessage);
}

// Penggunaan dengan pattern matching
void handleResponse(ApiResponse response) {
  switch (response) {
    case Success(message: var msg):
      print("Success: \$msg");
    case Error(errorMessage: var err):
      print("Error: \$err");
  }
}

void main() {
  handleResponse(Success("Data berhasil dimuat"));
  handleResponse(Error("Koneksi gagal"));
}

4. PRAKTIK SEALED CLASS: AppState Handler

Sealed class sangat ideal untuk merepresentasikan state aplikasi (Loading, Success, Error) dengan jaminan exhaustive checking. 


// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

sealed class AppState {}

class LoadingState extends AppState {}

class SuccessState extends AppState {
  final List items;
  SuccessState(this.items);
}

class ErrorState extends AppState {
  final String message;
  ErrorState(this.message);
}

void updateUI(AppState state) {
  switch (state) {
    case LoadingState():
      print('⏳ Loading...');
    case SuccessState(items: final items):
      print('✅ Success! Items:');
      for (var i = 0; i < items.length; i++) {
        print('   \${i + 1}. \${items[i]}');
      }
    case ErrorState(message: final msg):
      print('❌ Error: \$msg');
  }
}

void main() {
  final states = [
    LoadingState(),
    SuccessState(['Item 1', 'Item 2', 'Item 3']),
    ErrorState('Koneksi gagal'),
  ];
  
  for (var state in states) {
    updateUI(state);
    print('---');
  }
}

5. BASE CLASS - Hanya Bisa di-Extend

Base class memastikan bahwa class hanya bisa di-extend (menggunakan extends), tetapi tidak bisa di-implement (menggunakan implements) dari luar library. Berguna untuk framework atau base class yang ingin mengontrol inheritance.

📌 Fitur Base Class:
✅ Bisa di-extend dari luar library
❌ Tidak bisa di-implement dari luar library
✅ Memastikan pewarisan yang konsisten 

// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

// libbase.dart
base class Animal {
  void speak() {
    print('Animal speaks');
  }
}

// testbase.dart
import 'libbase.dart';

base class Dog extends Animal {
  @override
  void speak() {
    print('Woof!');
  }
}

base class Cat extends Animal {
  @override
  void speak() {
    print('Meow!');
  }
}

// ERROR: Tidak bisa implement base class dari luar library
// base class BigAnimal implements Animal { ... }

void main() {
  Animal dog = Dog();
  dog.speak(); // Woof!
  
  Animal cat = Cat();
  cat.speak(); // Meow!
}

6. FINAL Class - Tertutup untuk Ekstensi

Final class adalah class yang tidak bisa di-extend atau di-implement dari luar library. Cocok untuk utility class atau class yang ingin "ditutup" untuk modifikasi. 


// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

// libfinal.dart
final class Settings {
  final String theme;
  final int fontSize;
  
  Settings({required this.theme, required this.fontSize});
  
  void showSettings() {
    print("Theme: \${theme}, Font Size: \${fontSize}");
  }
}

// Bisa di-extend di dalam library yang sama
base class CustomSettings extends Settings {
  CustomSettings(String theme, int fontSize) 
      : super(theme: theme, fontSize: fontSize);
}

void main() {
  Settings settings = Settings(theme: "Dark", fontSize: 14);
  settings.showSettings(); // Theme: Dark, Font Size: 14
  
  CustomSettings custom = CustomSettings("Light", 16);
  custom.showSettings(); // Theme: Light, Font Size: 16
}

// ERROR di file terpisah:
// class NewSettings extends Settings {} // Tidak bisa!

7. Perbandingan Class Modifiers

Modifier Extend di luar Implement di luar Extend di dalam Implement di dalam
sealed❌❌✅✅
base✅❌✅✅
final❌❌✅✅
(default)✅✅✅✅
📌 Best Practices:
• Gunakan sealed untuk state/result yang terbatas
• Gunakan base untuk framework/base class
• Gunakan final untuk utility class

8. Latihan: Repository & MathUtils

Soal Latihan:
1. Buat base class Repository dengan method CRUD
2. Buat final class MathUtils dengan static methods 


// Nama: Putra Dhafir Nashrullah | NIM: 251410009 | Kelas: SI2A

// 1. Base class Repository
base class Repository<T> {
  final List<T> _items = [];
  
  void create(T item) {
    _items.add(item);
    print('Created: \$item');
  }
  
  List<T> readAll() {
    return List.unmodifiable(_items);
  }
  
  bool update(int index, T newItem) {
    if (index >= 0 && index < _items.length) {
      _items[index] = newItem;
      print('Updated index \$index: \$newItem');
      return true;
    }
    print('Update failed: index \$index out of range');
    return false;
  }
  
  bool delete(int index) {
    if (index >= 0 && index < _items.length) {
      var removed = _items.removeAt(index);
      print('Deleted: \$removed');
      return true;
    }
    print('Delete failed: index \$index out of range');
    return false;
  }
}

class UserRepository extends Repository<String> {
  void printAllUsers() {
    print('All users: \${readAll()}');
  }
}

// 2. Final class MathUtils
final class MathUtils {
  MathUtils._(); // Private constructor - tidak bisa di-instansiasi
  
  static int sum(List<int> numbers) {
    return numbers.fold(0, (a, b) => a + b);
  }
  
  static double average(List<int> numbers) {
    if (numbers.isEmpty) return 0;
    return sum(numbers) / numbers.length;
  }
  
  static int max(List<int> numbers) {
    if (numbers.isEmpty) throw Exception('List cannot be empty');
    return numbers.reduce((a, b) => a > b ? a : b);
  }
  
  static int min(List<int> numbers) {
    if (numbers.isEmpty) throw Exception('List cannot be empty');
    return numbers.reduce((a, b) => a < b ? a : b);
  }
  
  static bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= n ~/ 2; i++) {
      if (n % i == 0) return false;
    }
    return true;
  }
}

void main() {
  print('=== REPOSITORY DEMO ===');
  var userRepo = UserRepository();
  userRepo.create('Putra');
  userRepo.create('Dhafir');
  userRepo.create('Nashrullah');
  userRepo.printAllUsers();
  userRepo.update(1, 'Dhafir Update');
  userRepo.delete(0);
  userRepo.printAllUsers();
  
  print('\n=== MATHUTILS DEMO ===');
  var numbers = [10, 25, 7, 42, 18, 5];
  print('Numbers: \$numbers');
  print('Sum: \${MathUtils.sum(numbers)}');
  print('Average: \${MathUtils.average(numbers)}');
  print('Max: \${MathUtils.max(numbers)}');
  print('Min: \${MathUtils.min(numbers)}');
  print('Is 17 prime? \${MathUtils.isPrime(17)}');
  print('Is 25 prime? \${MathUtils.isPrime(25)}');
}
